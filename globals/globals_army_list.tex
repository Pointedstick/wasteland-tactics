% Formatting common to army lists
% ===============================

% Be landscape
\documentclass[landscape]{extarticle}

% Use tighter margins
\usepackage[margin=0.4in]{geometry}

% Don't number the pages
\pagenumbering{gobble}

% Use alternating row colors for the tables
\usepackage[table]{xcolor}
\definecolor{lightgray}{gray}{0.95}
\rowcolors{1}{white}{lightgray}

% Define some fixed-width columns
% Unit column
\newcolumntype{A}{>{\arraybackslash}m{2cm}}
% Weapons column
\newcolumntype{B}{>{\arraybackslash}m{10cm}}
% Special Rules column
\newcolumntype{C}{>{\arraybackslash}m{3.5cm}}
% Upgrades column
\newcolumntype{D}{>{\arraybackslash}m{4cm}}

% Inherit standard global formatting for everything else
\input{../globals/globals.tex}

% increase vertical spacing between table rows
\renewcommand{\arraystretch}{1.6}
