% Formatting common to book style documents
% ========================================

% Be a book with 12 pt font
% 'oneside' avoids blank pages after chapters
\documentclass[12pt, oneside]{book}

% Make page numbers show up on all pages
\pagestyle{plain}

% Use more normal print-style margins
\usepackage[margin=0.7in]{geometry}

% Use two columns
\usepackage{multicol}

% Allow hyperlinks
\usepackage{hyperref}

% Make hyperlinks look pretty
\usepackage{xcolor}
\hypersetup{
    colorlinks,
    linkcolor={black},
    urlcolor={blue}
}

% re-define TOC command to not have a leading page break for book format documents
\makeatletter
\newcommand*{\toccontents}{\@starttoc{toc}}
\makeatother

% Add a page border
% TODO: Find a better one
\usepackage{wallpaper}
\CenterWallPaper{1}{page_border_wood.jpg}

% Inherit standard global formatting for everything else
\input{globals/globals.tex}
